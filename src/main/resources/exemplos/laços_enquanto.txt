//Codigo para escrever uma sequencia de 1 ate 5

variaveis
	
  x:inteiro;

inicio
	
   x <- 1;

  //O enquanto só executara se a condicao for verdadeira
  enquanto x <= 5 faca //A verificacao é refeita a cada loop
    escrever x;
    x <- x+1;
  fim enquanto;

fim
