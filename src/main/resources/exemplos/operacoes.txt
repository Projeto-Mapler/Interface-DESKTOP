variaveis
	resultado: real;
inicio
    //ADICAO, definida como a soma de valores e para isso utiliza o operador "+"
	resultado <- 5.0 + 4.0;
	escrever "Operação: 5 + 4 = ", resultado;
	
	//SUBTRACAO, definida como a diferença entre valores e utiliza como operador o "-"
	resultado <- 5.0 - 4.0;
	escrever "Operação: 5 - 4 = ", resultado;	
		
	//MULTIPLICACAO, é o produto de valores e utiliza o operador "*"
	resultado <- 5.0 * 4.0; 
	escrever "Operação: 5 * 4 = ", resultado;
		
	//DIVISAO, a divisao utiliza um dividendo e um divisor para seu calculo e usa o operador "/"
	resultado <- 5.0 / 4.0;
	escrever "Operação: 5 / 4 = ",  resultado;	 
fim
