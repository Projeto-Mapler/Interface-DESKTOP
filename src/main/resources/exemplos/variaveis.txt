//VARIAVEIS são espaços da memoria reservados para armazenar valores obtidos pelo algoritmo
//Em portugol, as variaveis sao declaradas no parte do bloco principal chamada "VARIAVEIS" e depois usado
//o seu nome declarado para acessar e manipular seus valores

//SINTAXE
//	Uma variavel é declarada dentro da estrutura VARIAVEIS usando "nome_da_variavel: tipo_da_variavel;"
//	Uma variavel para ser acessada basta usar seu nome na estrutura INICIO, EXEMPLO: "escrever nome_da_variavel;"
//	Um variavek pode ser manipulada atribuindo valores, variaveis ou expressoes a essa variavel, EXEMPLO "nome_variavel <- VALOR"

//OBS
//	O sistema ira reportar erro sempre que uma variavel chamada nao tenha sido declarada  
//	É considerado como erro uma variavel receber um valor de tipagem diferente ao declarado para aquela variavel

//EXEMPLO

variaveis
	
  // numericos
  var_inteiro: inteiro;
  var_real: real;
  
  // booleano
  var_logico: logico;
  
  // literais
  var_cadeia: cadeia;
  var_caractere: caractere;

  // vetor
  var_vetor, var_vetor_2: vetor[1..4] de inteiro;

inicio
  
  var_inteiro <- -1;
  
  var_real <- 2.5;

  var_cadeia <- "São Luís";

  var_caractere <- "c";

  var_logico <- verdadeiro;

  var_vetor[1] <- 2;
   var_vetor_2[1] <- 2;

fim
