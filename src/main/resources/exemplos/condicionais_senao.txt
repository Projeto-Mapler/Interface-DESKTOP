//O COMANDO "SENAO" É UM COMPLEMENTO DO COMANDO "SE"

//SUA SINTAXE É: "se X entao <ACAO> senao <ACAO2> fim se;"
			                        //O senao sera executado sempre que a condicao X for falsa
			                        	  //A <ACAO2> serao comandos desejados caso X seja falso
			                        	  	      //O fim se deverá ficar apos os comandos do senao, indicando que o se e o senao estao em um mesmo bloco		             

//OBSERVACOES: O comando "SENAO" aceita outros blocos de "SE" dentro de sua <ACAO2>, mas estes devem ser fechados antes que o "SE" mais externo finalize.


//EXEMPLO
//	verificando a faixa etaria de idades presentes em um vetor


variaveis
	
  idades: vetor[1..4] de inteiro;
  i: inteiro;

inicio
	
  idades[1] <- 9;
  idades[2] <- 15;
  idades[3] <- 30;
  idades[4] <- 60;

  para i de 1 ate 4 passo 1 faca 
    se idades[i] > 11 e idades[i] < 18 entao 
      escrever "adolescente";
    senao 
      se idades[i] >= 18  e idades[i] < 60 entao 
        escrever "adulto";
       senao 
        se idades[i] >= 60 entao
          escrever "idoso";
        senao
          escrever "criança";
        fim se;
      fim se;
    fim se;
  fim para;

fim
