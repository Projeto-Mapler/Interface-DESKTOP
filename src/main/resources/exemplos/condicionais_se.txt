//O COMANDO "SE" É UTILIZADO DENTRO DO BLOCO INICIO

//SUA SINTAXE É: "se X entao <ACAO> fim se;"
                     //Sendo X uma condicao qualquer
                             //<ACAO> os comandos desejados que execute caso X seja verdadeiro
                                    //O comando "se" sempre deve ser finalizado com "fim se;"
                                    
//OBSERVACOES: O comando "SE" aceita outros blocos de "SE" dentro de sua <ACAO>, mas estes devem ser fechados antes que o "SE" mais externo finalize.

//EXEMPLO
//	Verificando se idades presentes em um vetor sao de pessoas adolescentes

variaveis
	
  idades: vetor[1..4] de inteiro;
  i: inteiro;

inicio
	
  idades[1] <- 9;
  idades[2] <- 15;
  idades[3] <- 30;
  idades[4] <- 60;

  para i de 1 ate 4 passo 1 faca 
    se idades[i] > 11 e idades[i] < 18 entao 
      escrever "adolescente";
      se idades[i] = 15 entao
          escrever "adolescente com 15 anos!";
      fim se; //finalizando o "se" inteno que verifica se idades é igual a 15
    fim se; //finalizando o primeiro "se" que verifica se é adolescente
  fim para;

fim
