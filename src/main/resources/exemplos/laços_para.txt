//Codigo para escrever uma sequencia de 1 ate 5, incrementando o valor de 1 a cada repeticao

variaveis
	
  x:inteiro;

inicio
	
   x <- 1;

  //O PARA só executara se a condicao for verdadeira
  para x de 1 ate 5 passo 1 faca //O passo indica o valor a ser incrementado na variavel a cada loop
		escrever x;
  fim para;

fim
