//LAÇOS DE REPETIÇÃO SAO BLOCOS QUE UTILIZAM CONDICOES PARA FICAR RODANDO VARIAS VEZES UM MESMO TRECHO DE CODIGO

//EXEMPLO DE CODIGO USANDO OS LACOS DE REPETICAO
variaveis
	
  x:inteiro;

inicio
	
  para x de 1 ate 5 passo 1 faca 
		escrever x;
  fim para;

  escrever " "; // escreve linha em branco
  x<-1;

  enquanto x <= 5 faca 
    escrever x;
    x<- x+1;
  fim enquanto;

  
  escrever " "; // escreve linha em branco
  x<-1;

  repita 
    escrever x;
    x<- x+1;
  ate (x <= 5);

fim



// PARA

//	O laço "PARA" coloca o valor de um parametro em uma variavel qualquer e depois autoincrementa seu valor até o valor de seu segundo parametro, 
//	o bloco irá finalizar quando o valor da variavel sair do intervalo dos parametros
//SINTAXE PARA
//	"para X de N1 ate N2 faca <ACAO> passo Y fim para;"
//	X é uma variavel qualquer
//	N1 é o parametro que define o valor inicial de X, esse valor será atribuido automaticamente pelo laço para a variavel.
//	N2 é o parametro que define o valor final de X, onde no momento em que X sair do intervalo de N1 até N2 o laço será finalizado
//	<ACAO> é o conjunto de comandos que o bloco irá executar
//	Y é o valor que sera incrementado em X cada vez o laço terminar seu bloco interno
//OBS PARA
//	quando N1 < N2, o laço irá incrementar o valor de X
//	quando N1 = N2, o laço irá rodar uma unica vez
//	quando N1 > N2, o laço não executará
//	deve ser sempre finalizado com o fim para
//	um bloco PARA aceita outros blocos PARA internos, ao fechar um PARA interno, nao finalizara o PARA externo

//EXEMPLO PARA

variaveis
	
  x:inteiro;

inicio
	
   x <- 1;

  para x de 1 ate 5 passo 1 faca 
		escrever x;
  fim para;

fim




// ENQUANTO
//	O laço "enquanto" utiliza uma condiçao para repetir seu bloco interno, diferente do "PARA" este nao realiza autoincremento
//	O laço ira finalizar quando a condição nao for verdadeira
//SINTAXE ENQUANTO
//	"enquanto X faca <ACAO> fim enquanto;"
//	X é uma condicao qualquer
//	<ACAO> é o conjunto de comandos que o bloco irá executar
//OBS
//	um bloco ENQUANTO aceita outros blocos ENQUANTO internos, ao fechar um ENQUANTO interno, nao finalizara o ENQUANTO externo
//EXEMPLO ENQUANTO
//	Codigo para escrever uma sequencia de 1 ate 5

variaveis
	
  x:inteiro;

inicio
	
   x <- 1;

  enquanto x <= 5 faca 
    escrever x;
    x <- x+1;
  fim enquanto;

fim




// REPITA
//	O laço "repita" funciona de forma semelhante ao laço "enquanto", porem a condição é testada apenas ao fim da execução de seu bloco interno de comandos
//	O laço ira finalizar quando a condição nao for verdadeira
//SINTAXE REPITA
//	"repita <ACAO> ate X;"
//	<ACAO> é o conjunto de comandos que o bloco irá executar
//	X é uma condicao qualquer
//OBS
//	Por conta de seu bloco executar antes da condiçao, o bloco executara pelo menos uma vez, mesmo que a condicao seja sempre falsa
//	um bloco REPITA aceita outros blocos REPITA internos, ao fechar um REPITA interno, nao finalizara o REPITA externo
//EXEMPLO REPITA
//	Codigo para escrever uma sequencia de 1 ate 5

variaveis
	
  x:inteiro;

inicio
	
  x<-1;

  repita 
    escrever x;
    x<- x+1;
  ate (x <= 5);

fim
