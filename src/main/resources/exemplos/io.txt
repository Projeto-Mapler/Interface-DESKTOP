//ALGORITMO
//Lendo nome do usuario e em seguida dando Ola

variaveis

  var_nome: cadeia;

inicio

  //A cadeia de caracteres passada em texto sera mostrada no terminal para o usuario
  escrever "Digite seu nome: ";
  
  //O comando usado receber dados do usuario é o "ler"
  ler var_nome; //o valor digitado pelo usuario será salvo na variavel associada ao comando ler
  
  //A uniao da cadeia de caracteres passada em texto e do valor presente em "variavel" sera mostrada no terminal para o usuario
  escrever "Olá ", var_nome;

fim
